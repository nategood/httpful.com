 53:     /**
 54:      * Parse the response into a clean data structure
 55:      * (most often an associative array) based on the expected
 56:      * Mime type.
 57:      * @return array|string|object the response parse accordingly
 58:      * @param string Http response body
 59:      */
 60:     public function _parse($body)
 61:     {
 62:         // If the user decided to forgo the automatic
 63:         // smart parsing, short circuit.
 64:         if (!$this->request->auto_parse) {
 65:             return $body;
 66:         }
 67: 
 68:         // If provided, use custom parsing callback
 69:         if (isset($this->request->parse_callback)) {
 70:             return call_user_func($this->request->parse_callback, $body);
 71:         }
 72: 
 73:         // Use the Content-Type from the response if we didn't explicitly 
 74:         // specify one as part of our `Request`
 75:         $parse_with = (empty($this->request->expected_type) && isset($this->content_type)) ?
 76:             $this->content_type :
 77:             $this->request->expected_type;
 78: 
 79:         switch ($parse_with) {
 80:             case Mime::JSON:
 81:                 $parsed = json_decode($body, false);
 82:                 if (!$parsed) throw new \Exception("Unable to parse response as JSON");
 83:                 break;
 84:             case Mime::XML:
 85:                 $parsed = simplexml_load_string($body);
 86:                 if (!$parsed) throw new \Exception("Unable to parse response as XML");
 87:                 break;
 88:             case Mime::FORM:
 89:                 $parsed = array();
 90:                 parse_str($body, $parsed);
 91:                 break;
 92:             default:
 93:                 $parsed = $body;
 94:         }
 95:         return $parsed;
 96:     }

114:     public function _parseCode($headers)
115:     {
116:         $parts = explode(' ', substr($headers, 0, strpos($headers, "\n")));

118:             throw new \Exception("Unable to parse response code from HTTP response due to malformed response");

127:     public function _interpretHeaders()
128:     {
129:         // Parse the Content-Type and charset
130:         $content_type = explode(';', $this->headers['Content-Type']);
131: 
132:         $this->content_type = $content_type[0];
133:         if (count($content_type) == 2 && strpos($content_type[1], '=') !== false) {
134:             list($nill, $this->charset) = explode('=', $content_type[1]);
135:         }
136:         // RFC 2616 states "text/*" Content-Types should have a default
137:         // charset of ISO-8859-1. "application/*" and other Content-Types
138:         // are assumed to have UTF-8 unless otherwise specified.
139:         // http://www.w3.org/Protocols/rfc2616/rfc2616-sec3.html#sec3.7.1
140:         // http://www.w3.org/International/O-HTTP-charset.en.php
141:         if (!isset($this->charset)) {
142:             $this->charset = substr($this->content_type, 5) === 'text/' ? 'iso-8859-1' : 'utf-8';
143:         }
144:     }

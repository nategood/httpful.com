660:     /**

662:      * library cURL to set up the HTTP request.
663:      * Note: It does NOT actually send the request
664:      * @return Request $this;
665:      */
666:     public function _curlPrep()
667:     {
668:         // Check for required stuff
669:         if (!isset($this->uri))
670:             throw new \Exception('Attempting to send a request before defining a URI endpoint.');
671: 
672:         $ch = curl_init($this->uri);
673: 
674:         curl_setopt($ch, CURLOPT_CUSTOMREQUEST, $this->method);
675: 
676:         if ($this->hasBasicAuth()) {
677:             curl_setopt($ch, CURLOPT_USERPWD, $this->username . ':' . $this->password);
678:         }
679: 
680:         if ($this->hasClientSideCert()) {
681: 
682:             if (!file_exists($this->client_key))
683:                 throw new \Exception('Could not read Client Key');
684: 
685:             if (!file_exists($this->client_cert))
686:                 throw new \Exception('Could not read Client Certificate');
687: 
688:             curl_setopt($ch, CURLOPT_SSLCERTTYPE,   $this->client_encoding);
689:             curl_setopt($ch, CURLOPT_SSLKEYTYPE,    $this->client_encoding);
690:             curl_setopt($ch, CURLOPT_SSLCERT,       $this->client_cert);
691:             curl_setopt($ch, CURLOPT_SSLKEY,        $this->client_key);
692:             curl_setopt($ch, CURLOPT_SSLKEYPASSWD,  $this->client_passphrase);
693:             // curl_setopt($ch, CURLOPT_SSLCERTPASSWD,  $this->client_cert_passphrase);
694:         }
695: 
696:         if ($this->follow_redirects) {
697:             curl_setopt($ch, CURLOPT_FOLLOWLOCATION, true);
698:             curl_setopt($ch, CURLOPT_MAXREDIRS, $this->max_redirects);
699:         }
700: 
701:         curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, $this->strict_ssl);
702:         curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
